def convert_data(
    df: pd.DataFrame,
    reorder_cols=[],
    date_cols=[],
    numeric_cols=[],
    infer_numeric=False,
    replace_values=(),
) -> pd.DataFrame:
    """
    Apply conversions on a dataframe.

    Args:
        df: The dataframe to be converted.
        reorder_cols: A list of columns in the desired order, to replace the existing
            column order in the given dataframe.
        date_cols: A list of columns to be converted to date format.
        numeric_cols: A list of columns to be converted to numeric format.
        infer_numeric: A boolean indicating whether or not to infer numeric columns.
        replace_values: A tuple where (or a list of tuples, where for each tuple),
            the first element is to be replaced with the second element.

    Returns:
        pd.DataFrame
    """
    # reorder columns
    if reorder_cols:
        df = df[reorder_cols]
    # replace values
    if replace_values:
        if isinstance(replace_values, tuple):
            df = df.apply(
                lambda x: x.replace(replace_values[0], replace_values[1]), axis=0
            )
        elif isinstance(replace_values, list):
            for pair in replace_values:
                df = df.apply(lambda x: x.replace(pair[0], pair[1]), axis=0)
    for col_id in df.columns:
        # convert given columns to numeric format
        if col_id in numeric_cols:
            df[col_id] = pd.to_numeric(df[col_id], errors="coerce")
        # infer which columns to convert to numeric format
        if infer_numeric:
            try:
                df[col_id] = pd.to_numeric(df[col_id])
            except:
                pass
        # convert given columns to date format
        if col_id in date_cols:
            df[col_id] = pd.to_datetime(df[col_id], errors="coerce")

    return df

def filter_data(df: pd.DataFrame, filters: list, join: str = "AND") -> pd.DataFrame:
    """
    Apply filters to a dataframe.

    Args:
        df: The dataframe to be filtered.
        filters: A list of filters. This outermost list may be nested to contain more lists of
            filters. Each filter is described as a dictionary with the key "colname" and at
            least one of the filter type keys ["isbetween", "isin", "contains", "isnull"]. For any dictionary
            that contains more than one filter type key, the filters are joined by "OR".
        join: The logical operator ["AND", "OR"] for joining outermost list of filters. The logical
            operator is recursively flipped for any nested list. For example, for a list is joined by
            "AND", a nested list will be joined by "OR", and a nested list within this nested lists
            will be joined by "AND".

    Returns:
        pd.DataFrame
    """

    def get_init_mask(df: pd.DataFrame, join: str = "AND") -> pd.Series:
        """
        Create an initial mask matching the length of the dataframe. This mask will either contain
        all True or all False, depending on if the join is AND or OR respectively, since booleans
        have these properties: True & A = A, False | A = A
        """
        start_bool = True if join == "AND" else False
        init_mask = pd.Series(np.repeat(start_bool, df.shape[0]), index=df.index)
        return init_mask

    def get_mask(df: pd.DataFrame, filters: list, join: str = "AND") -> pd.Series:
        """
        Compute a mask combining the (nested) list of filters. For any nested list, this function
        is recursively called.
        """
        final_mask = get_init_mask(df, join)

        for item in filters:
            if isinstance(item, dict):
                if not "colname" in item:
                    raise Exception(
                        f"Filter dictionary {item} must have 'colname' defined as a key."
                    )

                valid_filter_types = ["isbetween", "isin", "contains", "isnull"]
                valid_filter_flag = any(
                    filter_type in item for filter_type in valid_filter_types
                )
                if not valid_filter_flag:
                    raise Exception(
                        f"Filter dictionary {item} must have one of {valid_filter_types} defined as a key."
                    )

                mask = get_init_mask(df, "OR")
                apply_filter_flag = False

                if "isbetween" in item:
                    apply_filter_flag = True
                    range_mask = get_init_mask(df, "AND")
                    start = item["isbetween"][0]
                    end = item["isbetween"][1]
                    # exclude null to apply range without hitting any nulls
                    if start or end:
                        range_mask = range_mask & (~df[item["colname"]].isnull())
                    # convert date string if date column is of type datetime
                    if is_datetime(df[item["colname"]]):
                        if isinstance(start, str):
                            start = pd.to_datetime(start)
                        if isinstance(end, str):
                            end = pd.to_datetime(end)
                    # apply masks if defined
                    if start is not None:
                        range_mask = range_mask & (df[item["colname"]] >= start)
                    if end is not None:
                        range_mask = range_mask & (df[item["colname"]] <= end)
                    mask = mask | range_mask

                # if "isin" in item and item["isin"]:
                if "isin" in item and item["isin"] is not None:
                    apply_filter_flag = True
                    mask = mask | df[item["colname"]].isin(item["isin"])

                if "contains" in item and item["contains"]:
                    apply_filter_flag = True
                    mask = mask | df[item["colname"]].str.contains(
                        r"\b(?:{})\b".format("|".join(item["contains"]))
                    )

                if "isnull" in item and item["isnull"]:
                    apply_filter_flag = True
                    mask = mask | df[item["colname"]].isnull()

                if not apply_filter_flag:
                    mask = get_init_mask(df, "AND")

            elif isinstance(item, list):
                nested_join = "OR" if join == "AND" else "AND"
                mask = get_mask(df, item, join=nested_join)

            else:
                raise Exception(
                    f"Filter item {item} is of invalid type '{type(item)}'."
                )

            final_mask = (final_mask & mask) if join == "AND" else (final_mask | mask)

        return final_mask

    mask = get_mask(df, filters, join)
    df = df.loc[mask]
    return df

def pagination_v1(df, page_no, limit):
    """
    implements pagination by chunking the dataframe
    """
    error_json = {"Message": "Page number {} not found.".format(page_no)}
    df_len = len(df)

    if df_len / limit > page_no:
        if df_len >= limit:
            df = get_np_array(df, n)
            try:
                tbl_json = df[page_no].apply(lambda x: json.loads(x.to_json()), axis=1)

                pass

            current_row_count = len(df)
            n = 1
        try:
            res = {
                "current_row_count": current_row_count,
                "row_count": df_len,
                "total_pages": n,
                "tbl": tbl_json,
            }
        except Exception:
            res = error_json
    else:
        res = error_json
    return res

def pivot_data(df, values, index, columns, aggregation, margins):
    """
    Method to pivot the data as per requirement.

    Args:
        df (dataframe): _description_
        values: Numeric column which is root of pivoting
        index (_type_): _description_
        columns (list): columns that remain as row
        aggregation : Aggregation value, (Ex: sum or count)
        margins (boolean): If 'margins=True', special All columns and rows will be added with partial group
        aggregates across the categories on the rows and columns.

    Returns:
        dataframe: Method to pivot the data as per requirement.
    """

    if aggregation == "sum":
        aggregation = np.sum

    pivoted_df = pd.pivot_table(
        df,
        values=values,
        index=index,
        columns=columns,
        aggfunc=aggregation,
        margins=margins,
    )
    return pivoted_df

def sunburst(df, L1, L2, basedon):
    """
    converts the data as per UI requirement to plot the sunburst graph.

    Args:
        df (_type_): _description_
        L1 (_type_): _description_
        L2 (_type_): _description_
        basedon (_type_): _description_

    Returns:
        _type_: _description_
    """
    df = df[[L1, basedon, L2]]
    df = convert_data(df=df, numeric_cols=[basedon])
    df = df.groupby([L1, L2])[basedon].agg(["sum", "count"]).reset_index()
    df.rename(columns={"sum": basedon}, inplace=True)
    df = df[[L1, L2, basedon, "count"]]
    df[L1] = df[L1].replace("", "N/A")
    df[L2] = df[L2].replace("", "N/A")
    df.rename(columns={L1: "root", L2: "leaf"}, inplace=True)
    return df

def unshuffle(list1, list2):
    """
    shuffles list2 based on list1
    """
    res = [x for _, x in sorted(zip(list2, list1))]
    return "".join(res)
