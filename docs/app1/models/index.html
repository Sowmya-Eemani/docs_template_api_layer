class APIPerformance(models.Model):
    """
    Measures the API performance

    Args:
        models (_type_): _description_
    """

    ID = models.AutoField(primary_key=True)
    BRANCH = models.CharField(max_length=20, default="Dev")
    FUNCTION_NAME = models.CharField(max_length=100)
    FUNCTION_ARGS = models.CharField(max_length=1024)
    FUNCTION_PATH = models.CharField(max_length=1024, null=True, blank=True)
    START_TIME = models.DateTimeField(default=timezone.now)
    END_TIME = models.DateTimeField(default=timezone.now)
    DURATION = models.FloatField()
    ERROR_MESSAGE = models.CharField(max_length=1024, default="No Errors")
